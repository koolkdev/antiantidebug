DEFINE_GROUP UPDATE_MATH + - ^ & |

DEFINE_GROUP FIELDS VMStructFieldByte VMStructFieldWord VMStructFieldDword

### Decode ###
DEFINE_EXPRESSION_TEMPLATE
($[X] $G[SIMPLE_MATH:OP] VMStructFieldDword(?O[KEY_*:KEY]))
=>
DecodeWithKey($[X], SimpleOperation(Operation($[OP]), VMStructFieldDword(?O[KEY])))

(ReadParameterWord($N[OFFSET]) $G[SIMPLE_MATH:OP] $N[NUMBER])
=>

# KEY_DECODE

(DecodeWithKey(ReadParameterWord($N[OFFSET]), SimpleOperation($[OP1], VMStructFieldDword(?O[KEY_*:KEY]))) $G[SIMPLE_MATH:OP2] $N[NUMBER])

DecodeWithNumber(DecodeWithKey(ReadParameterWord($N[OFFSET]), SimpleOperation($[OP1], VMStructFieldDword(?O[KEY]))), SimpleOperation(Operation($[OP2]), $N[NUMBER]))

### Key Updates ###
DEFINE_TEMPLATE
VMStructFieldDword(?O[KEY_*:KEY]) $G[UPDATE_MATH:OP]= $N[NUMBER]
=>
UpdateKey(VMStructFieldDword(?O[KEY]), SimpleOperation(Operation($[OP]), $[NUMBER]))

# KEY_DECODE
DEFINE_TEMPLATE
VMStructFieldDword(?O[KEY_*:KEY]) $G[UPDATE_MATH:OP]= $V[VAR]
=>
UpdateKeyDecode(VMStructFieldDword(?O[KEY]), SimpleOperation(Operation($[OP]), $[VAR]))

# KEY_COND (0x1), KEY_DECODE (0x2)
DEFINE_TEMPLATE
If(((VMStructFieldDword(?O[KEY_*:KEY]) & $N[BIT]) != 0x0))
=>
UpdateKeyCond(VMStructFieldDword(?O[KEY]), $[BIT], None, None)

# KEY_COND (0x1), KEY_DECODE (0x2)
DEFINE_TEMPLATE
If(((VMStructFieldDword(?O[KEY_*:KEY]) & $N[BIT]) != 0x0))
    UpdateKey(VMStructFieldDword(?O[KEY_*:KEY2]), $[X])
=>
UpdateKeyCond(VMStructFieldDword(?O[KEY]), $[BIT], UpdateKey(VMStructFieldDword(?O[KEY2]), $[X]), None)

# KEY_DECODE (0x2)
DEFINE_TEMPLATE
If(((VMStructFieldDword(?O[KEY_*:KEY]) & $N[BIT]) != 0x0))
    UpdateKey(VMStructFieldDword(?O[KEY_*:KEY2]), $[X])
    UpdateKey(VMStructFieldDword(?O[KEY_*:KEY3]), $[Y])
=>
UpdateKeyCond(VMStructFieldDword(?O[KEY]), $[BIT], UpdateKey(VMStructFieldDword(?O[KEY2]), $[X]), UpdateKey(VMStructFieldDword(?O[KEY3]), $[Y]))

DEFINE_TEMPLATE
VMStructFieldDword(?O[KEY_*:KEY]) $G[UPDATE_MATH:OP]= (VMStructFieldDword(?O[KEY_*:KEY2]) $G[UPDATE_MATH:OP2] $N[NUMBER])
=>
UpdateKeyComplex(VMStructFieldDword(?O[KEY]), Operation($[OP]), VMStructFieldDword(?O[KEY2]), SimpleOperation(Operation($[OP2]), $N[NUMBER]))

DEFINE_TEMPLATE
VMStructFieldDword(?O[KEY_*:KEY]) $G[UPDATE_MATH:OP]= VMStructFieldDword(?O[KEY_*:KEY2])
=>
UpdateKeyComplex(VMStructFieldDword(?O[KEY]), Operation($[OP]), VMStructFieldDword(?O[KEY2]), None)
