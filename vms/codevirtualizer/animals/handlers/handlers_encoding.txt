OPTION DONT_OPTIMIZE

DEFINE_GROUP SIMPLE_MATH + - ^
DEFINE_GROUP UPDATE_MATH + - ^ & |

DEFINE_GROUP FIELDS VMStructFieldByte VMStructFieldWord VMStructFieldDword

### Decode ###
DEFINE_EXPRESSION_TEMPLATE
($[X] $G[SIMPLE_MATH:OP] VMStructFieldDword(?O[KEY_*:KEY]))
=>
DecodeWithKey($[X], SimpleOperation(Operation($[OP]), VMStructFieldDword(?O[KEY])))

# For decoding next handler
DEFINE_EXPRESSION_TEMPLATE
(ReadParameterWord($N[OFFSET]) $G[SIMPLE_MATH:OP] $N[NUMBER])
=>
DecodeWithNumber(ReadParameterWord($N[OFFSET]), SimpleOperation(Operation($[OP]), $N[NUMBER]))

DEFINE_EXPRESSION_TEMPLATE
(DecodeWithKey(ReadParameterWord($N[OFFSET]), SimpleOperation($[OP1], VMStructFieldDword(?O[KEY_DECODE]))) $G[SIMPLE_MATH:OP2] $N[NUMBER])
=>
DecodeWithNumber(DecodeWithKey(ReadParameterWord($N[OFFSET]), SimpleOperation($[OP1], VMStructFieldDword(?O[KEY_DECODE]))), SimpleOperation(Operation($[OP2]), $N[NUMBER]))

### Key Updates ###
DEFINE_TEMPLATE
VMStructFieldDword(?O[KEY_*:KEY]) $G[UPDATE_MATH:OP]= $N[NUMBER]
=>
UpdateKey(VMStructFieldDword(?O[KEY]), SimpleOperation(Operation($[OP]), $[NUMBER]))

DEFINE_TEMPLATE
VMStructFieldDword(?O[KEY_DECODE*:KEY]) $G[UPDATE_MATH:OP]= $V[VAR]
=>
UpdateKeyDecode(VMStructFieldDword(?O[KEY]), SimpleOperation(Operation($[OP]), $[VAR]))

DEFINE_TEMPLATE
If(((VMStructFieldDword(?O[KEY_COND]) & 0x1) != 0x0))
=>
UpdateKeyCond(None)

DEFINE_TEMPLATE
If(((VMStructFieldDword(?O[KEY_COND]) & 0x1) != 0x0))
    UpdateKey(VMStructFieldDword(?O[KEY_COND]), $[X])
=>
UpdateKeyCond($[X])

DEFINE_TEMPLATE
If(((VMStructFieldDword(?O[KEY_DECODE]) & 0x2) != 0x0))
=>
UpdateKeyCond2(None, None)

DEFINE_TEMPLATE
If(((VMStructFieldDword(?O[KEY_DECODE]) & 0x2) != 0x0))
    UpdateKey(VMStructFieldDword(?O[KEY_COND]), $[Y])
=>
UpdateKeyCond2(None, $[Y])

DEFINE_TEMPLATE
If(((VMStructFieldDword(?O[KEY_DECODE]) & 0x2) != 0x0))
    UpdateKey(VMStructFieldDword(?O[KEY_DECODE]), $[X])
=>
UpdateKeyCond2($[X], None)

DEFINE_TEMPLATE
If(((VMStructFieldDword(?O[KEY_DECODE]) & 0x2) != 0x0))
    UpdateKey(VMStructFieldDword(?O[KEY_DECODE]), $[X])
    UpdateKey(VMStructFieldDword(?O[KEY_COND]), $[Y])
=>
UpdateKeyCond2($[X], $[Y])

DEFINE_TEMPLATE
If(((VMStructFieldDword(?O[KEY_DECODE]) & 0x2) != 0x0))
    UpdateKey(VMStructFieldDword(?O[KEY_DECODE]), $[X])
    UpdateKey(VMStructFieldDword(?O[KEY_COND]), $[Y])
=>
UpdateKeyCond2($[X], $[Y])

DEFINE_TEMPLATE
VMStructFieldDword(?O[KEY_DECODE]) $G[UPDATE_MATH:OP]= (VMStructFieldDword(?O[KEY_COND]) $G[UPDATE_MATH:OP2] $N[NUMBER])
=>
UpdateKeyDecodeCond(Operation($[OP]), SimpleOperation(Operation($[OP2]), $N[NUMBER]))

DEFINE_TEMPLATE
VMStructFieldDword(?O[KEY_DECODE]) $G[UPDATE_MATH:OP]= VMStructFieldDword(?O[KEY_COND])
=>
UpdateKeyDecodeCond(Operation($[OP]), None)

DEFINE_TEMPLATE
VMStructFieldDword(?O[KEY_COND]) $G[UPDATE_MATH:OP]= (VMStructFieldDword(?O[KEY_DECODE]) $G[UPDATE_MATH:OP2] $N[NUMBER])
=>
UpdateKeyDecodeCond2(Operation($[OP]), SimpleOperation(Operation($[OP2]), $N[NUMBER]))

DEFINE_TEMPLATE
VMStructFieldDword(?O[KEY_COND]) $G[UPDATE_MATH:OP]= VMStructFieldDword(?O[KEY_DECODE])
=>
UpdateKeyDecodeCond2(Operation($[OP]), None)
