OPTION RUN_ONCE

DEFINE_MACRO[32] SIZE DWORD
DEFINE_MACRO[64] SIZE QWORD

DEFINE_GROUP MEMS_1 REG
DEFINE_GROUP MEMS_2 REGREG REGMUL
DEFINE_GROUP MEMS_3 REGREGMUL

# [REG*X]
DEFINE_TEMPLATE
MOV_@SIZE@_VAR_REG V X
SHL_@SIZE@_VAR_IMM V Y
=>
MOV_@SIZE@_VAR_REGMUL V X (1<<Y)

# [REG+REG]
DEFINE_TEMPLATE
MOV_@SIZE@_VAR_REG V X
ADD_@SIZE@_VAR_REG V Y
=>
MOV_@SIZE@_VAR_REGREG V X Y

# [REG+IMM]
DEFINE_TEMPLATE
MOV_@SIZE@_VAR_REG V X
ADD_@SIZE@_VAR_IMM V Y
=>
MOV_@SIZE@_VAR_REGIMM V X Y

# [REG*X+REG]
DEFINE_TEMPLATE
MOV_@SIZE@_VAR_REGMUL V Y Z
ADD_@SIZE@_VAR_REG V X
=>
MOV_@SIZE@_VAR_REGREGMUL V X Y Z

# [REG*X+IMM]
DEFINE_TEMPLATE
MOV_@SIZE@_VAR_REGMUL V X Y
ADD_@SIZE@_VAR_IMM V Z
=>
MOV_@SIZE@_VAR_REGMULIMM V X Y Z

# [REG+REG+IMM]
DEFINE_TEMPLATE
MOV_@SIZE@_VAR_REGREG V X Y
ADD_@SIZE@_VAR_IMM V Z
=>
MOV_@SIZE@_VAR_REGREGIMM V X Y Z

# [REG+REG*MUL+IMM]
DEFINE_TEMPLATE
MOV_@SIZE@_VAR_REGREGMUL V X Y Z
ADD_@SIZE@_VAR_IMM V A
=>
MOV_@SIZE@_VAR_REGREGMULIMM V X Y Z A

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_IMM V A
ADD_VAR_BASEADDRESS V B
=>
MOV_@SIZE@_VAR_RELIMM V (A+B)

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS_1|IMM V X A
ADD_VAR_BASEADDRESS V B
=>
MOV_@SIZE@_VAR_*MEM*RELIMM V X (A+B)

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS_2|IMM V X Y A
ADD_VAR_BASEADDRESS V B
=>
MOV_@SIZE@_VAR_*MEM*RELIMM V X Y (A+B)

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS_3|IMM V X Y Z A
ADD_VAR_BASEADDRESS V B
=>
MOV_@SIZE@_VAR_*MEM*RELIMM V X Y Z (A+B)

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_RELIMM V X
ADD_@SIZE@_VAR_IMM V Y
=>
MOV_@SIZE@_VAR_RELIMM V (X+Y)

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_RELIMM V X
SUB_@SIZE@_VAR_IMM V Y
=>
MOV_@SIZE@_VAR_RELIMM V (X-Y)
