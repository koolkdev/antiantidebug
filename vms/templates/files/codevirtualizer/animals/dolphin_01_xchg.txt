OPTION RUN_ONCE

DEFINE_GROUP MEMS_1 IMM RELIMM
DEFINE_GROUP MEMS_2 REGREG REGMUL REGIMM REGRELIMM
DEFINE_GROUP MEMS_3 REGREGIMM REGREGRELIMM REGREGMUL REGMULIMM REGMULRELIMM
DEFINE_GROUP MEMS_4 REGREGMULIMM REGREGMULRELIMM

DEFINE_GROUP SIZES BYTE WORD DWORD
DEFINE_GROUP[64] SIZES QWORD

DEFINE_MACRO[32] SIZE DWORD
DEFINE_MACRO[64] SIZE QWORD

DEFINE_TEMPLATE_MACRO XCHG_PART
MOV_@SIZE@_VAR_REG VAR2 Y
MOV_|&SIZE=^SIZES|_REG_MEMVAR Y VAR1
MOV_|&SIZE=^SIZES|_MEMVAR_VAR VAR1 VAR2

DEFINE_TEMPLATE_MACRO XCHG_PART_SP
MOV_@SIZE@_VAR_REG VAR2 Y
MOV_@SIZE@_REG_MEMVAR Y VAR1
MOV_@SIZE@_MEMVAR_VAR VAR1 VAR2

DEFINE_TEMPLATE
MOV_|&SIZE=^SIZES|_VAR_REG VAR Y
MOV_|&SIZE=^SIZES|_REG_REG Y X
MOV_|&SIZE=^SIZES|_REG_VAR X VAR
=>
XCHG_*SIZE*_REG_REG X Y

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_REG VAR Y
MOV_|&SIZE=^SIZES|_REG_MEMREG Y X
MOV_|&SIZE=^SIZES|_MEMREG_VAR X VAR
=>
XCHG_*SIZE*_MEMREG_REG X Y

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS_1 VAR1 X
@XCHG_PART
=>
XCHG_*SIZE*_MEM*MEM*_REG X Y

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS_2 VAR1 X Z
@XCHG_PART
=>
XCHG_*SIZE*_MEM*MEM*_REG X Z Y

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS_3 VAR1 X Z A
@XCHG_PART
=>
XCHG_*SIZE*_MEM*MEM*_REG X Z A Y

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS_4 VAR1 X Z A B
@XCHG_PART
=>
XCHG_*SIZE*_MEM*MEM*_REG X Z A B Y
