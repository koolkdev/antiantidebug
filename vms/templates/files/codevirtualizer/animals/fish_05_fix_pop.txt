OPTION MATCH_ONE
OPTION RUN_ONCE

DEFINE_GROUP MEM_1 MUL REG
DEFINE_GROUP MEM_2 REGMUL

DEFINE_MACRO[32] SIZE DWORD
DEFINE_MACRO[64] SIZE QWORD
DEFINE_MACRO[32] SIZE_NUM 0x4
DEFINE_MACRO[64] SIZE_NUM 0x8

DEFINE_TEMPLATE
POP_@SIZE@_MEMREGIMM SP IMM:(IMM!=@SIZE_NUM@)
=>
POP_@SIZE@_MEMREGIMM SP (IMM-@SIZE_NUM@)

DEFINE_TEMPLATE
POP_@SIZE@_MEMREGIMM SP IMM:(IMM==@SIZE_NUM@)
=>
POP_@SIZE@_MEMREG SP

DEFINE_TEMPLATE
POP_@SIZE@_MEMREG SP
=>
POP_@SIZE@_MEMREGIMM SP (-@SIZE_NUM@)

DEFINE_TEMPLATE
POP_@SIZE@_MEMREGREGIMM X SP IMM:(IMM!=@SIZE_NUM@)
=>
POP_@SIZE@_MEMREGREGIMM SP (IMM-@SIZE_NUM@)

DEFINE_TEMPLATE
POP_@SIZE@_MEMREGREGIMM X SP IMM:(IMM==@SIZE_NUM@)
=>
POP_@SIZE@_MEMREGREG X SP

DEFINE_TEMPLATE
POP_@SIZE@_MEMREGREG X SP
=>
POP_@SIZE@_MEMREGREGIMM X SP (-@SIZE_NUM@)

DEFINE_TEMPLATE
POP_@SIZE@_MEMREG|&MEM=^MEM_1|IMM SP X IMM:(IMM!=@SIZE_NUM@)
=>
POP_@SIZE@_MEMREG*MEM*IMM SP X (IMM-@SIZE_NUM@)

DEFINE_TEMPLATE
POP_@SIZE@_MEMREG|&MEM=^MEM_1|IMM SP X IMM:(IMM==@SIZE_NUM@)
=>
POP_@SIZE@_MEMREG*MEM* SP X

DEFINE_TEMPLATE
POP_@SIZE@_MEMREG|&MEM=^MEM_1 SP X
=>
POP_@SIZE@_MEMREG*MEM*IMM SP X (-@SIZE_NUM@)

DEFINE_TEMPLATE
POP_@SIZE@_MEMREG|&MEM=^MEM_2|IMM SP X Y IMM:(IMM!=@SIZE_NUM@)
=>
POP_@SIZE@_MEMREG*MEM*IMM SP X Y (IMM-@SIZE_NUM@)

DEFINE_TEMPLATE
POP_@SIZE@_MEMREG|&MEM=^MEM_2|IMM SP X Y IMM:(IMM==@SIZE_NUM@)
=>
POP_@SIZE@_MEMREG*MEM* SP X Y

DEFINE_TEMPLATE
POP_@SIZE@_MEMREG|&MEM=^MEM_2 SP X Y
=>
POP_@SIZE@_MEMREG*MEM*IMM SP X Y (-@SIZE_NUM@)
