# Can't run once because cases like that:
# MOV_DWORD_VAR_REGIMM 0x1D 0x97 0xFFFFFFF8
# MOV_DWORD_VAR_RELIMM 0x5B 0x3F68F49
# MOV_DWORD_MEMVAR_VAR 0x1D 0x5B
#OPTION RUN_ONCE

DEFINE_GROUP CALL_OP CALL JMP
DEFINE_GROUP ONE_ARG_OP PUSH POP INC DEC NEG NOT
DEFINE_GROUP PUSH_POP PUSH POP
DEFINE_GROUP TWO_ARG_OP MOV ADD SUB XOR AND OR SHL SHR ROL ROR RCL RCR CMP TEST IMUL XCHG
DEFINE_GROUP TWO_ARG_OP_SIZES MOVSX MOVZX

DEFINE_GROUP ARGS REG REGHIGH IMM RELIMM
DEFINE_GROUP MEMS IMM RELIMM REG REGREG REGMUL REGIMM REGRELIMM REGREGIMM REGREGMUL REGMULIMM REGMULRELIMM REGREGMULIMM REGREGMULRELIMM

DEFINE_GROUP ARGS_MOV MEMREG MEMVAR

# TODO: verify the possible args. I assume that we will have to fix the number first so that is enough
DEFINE_GROUP BASE_ARGS REG MEMREG MEMVAR
DEFINE_GROUP BASE_MATH MOV ADD SUB XOR OR AND CMP TEST

DEFINE_GROUP SIZES BYTE WORD DWORD
DEFINE_GROUP[64] SIZES QWORD

DEFINE_MACRO[32] SIZE DWORD
DEFINE_MACRO[64] SIZE QWORD

# TODO: For the case LEA_DWORD_REG_RELIMM use mov instead in x86
DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *Y
MOV_|&SIZE=^SIZES|_REG_VAR X V
=>
LEA_*SIZE*_REG_*MEM* X *Y

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *X
END_VM
&OP=^CALL_OP|_MEMVAR V
=>
END_VM
*OP*_MEM*MEM* *X

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *X
&OP=^ONE_ARG_OP|_|&SIZE=^SIZES|_MEMVAR V
=>
*OP*_*SIZE*_MEM*MEM* *X

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_RELIMM V X
&OP=^PUSH_POP|_|&SIZE=^SIZES|_VAR V
=>
*OP*_*SIZE*_RELIMM X

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *X
&OP=^TWO_ARG_OP|_|&SIZE=^SIZES|_MEMVAR_|&ARG=^ARGS V Y
=>
*OP*_*SIZE*_MEM*MEM*_*ARG* *X Y

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *Y
&OP=^TWO_ARG_OP|_|&SIZE=^SIZES|_|&ARG=^ARGS|_MEMVAR X V
=>
*OP*_*SIZE*_*ARG*_MEM*MEM* X *Y

# It isn't possible to use addresses like that in 64 bit, only with lea
DEFINE_TEMPLATE[32]
MOV_@SIZE@_VAR_RELIMM V Y
MOV_|&SIZE=^SIZES|_|&ARG=^ARGS_MOV|_VAR X V
=>
MOV_*SIZE*_*ARG*_RELIMM X Y

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *X
&OP=^TWO_ARG_OP_SIZES|_|&SIZE=^SIZES|_|&SIZE2=^SIZES|_MEMVAR_|&ARG=^ARGS V Y
=>
*OP*_*SIZE*_*SIZE2*_MEM*MEM*_*ARG* *X Y

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *Y
&OP=^TWO_ARG_OP_SIZES|_|&SIZE=^SIZES||_|&SIZE2=^SIZES|_|&ARG=^ARGS|_MEMVAR X V
=>
*OP*_*SIZE*_*SIZE2*_*ARG*_MEM*MEM* X *Y

DEFINE_TEMPLATE[64]
MOV_QWORD_VAR_IMM V Y:(Y>=((1<<64)-(1<<(32-1))))
&OP=^BASE_MATH|_QWORD_|&ARG=^BASE_ARGS|_VAR X V
=>
*OP*_QWORD_*ARG*_IMM X Y
