OPTION RUN_ONCE

DEFINE_GROUP CALL_OP CALL JMP
DEFINE_GROUP ONE_ARG_OP PUSH POP INC DEC NEG NOT
DEFINE_GROUP PUSH_POP PUSH POP
DEFINE_GROUP TWO_ARG_OP MOV ADD SUB XOR AND OR SHL SHR ROL ROR RCL RCR CMP TEST IMUL XCHG
DEFINE_GROUP TWO_ARG_OP_SIZES MOVSX MOVZX

DEFINE_GROUP ARGS REG REGHIGH IMM RELIMM
DEFINE_GROUP MEMS IMM RELIMM REG REGREG REGMUL REGIMM REGRELIMM REGREGIMM REGREGMUL REGMULIMM REGMULRELIMM REGREGMULIMM REGREGMULRELIMM

DEFINE_GROUP SIZES BYTE WORD DWORD
DEFINE_GROUP[64] SIZES QWORD

DEFINE_MACRO[32] SIZE DWORD
DEFINE_MACRO[64] SIZE QWORD

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *Y
MOV_|&SIZE=^SIZES|_REG_VAR X V
=>
LEA_*SIZE*_REG_*MEM* X *Y

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *X
END_VM
&OP=^CALL_OP|_MEMVAR V
=>
END_VM
*OP*_MEM*MEM* *X

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *X
&OP=^ONE_ARG_OP|_|&SIZE=^SIZES|_MEMVAR V
=>
*OP*_*SIZE*_MEM*MEM* *X

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_RELIMM V X
&OP=^PUSH_POP|_|&SIZE=^SIZES|_VAR V
=>
*OP*_*SIZE*_RELIMM X

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *X
&OP=^TWO_ARG_OP|_|&SIZE=^SIZES|_MEMVAR_|&ARG=^ARGS V Y
=>
*OP*_*SIZE*_MEM*MEM*_*ARG* *X Y

DEFINE_TEMPLATE
MOV_@SIZE@_VAR_|&MEM=^MEMS V *Y
&OP=^TWO_ARG_OP|_|&SIZE=^SIZES|_|&ARG=^ARGS|_MEMVAR X V
=>
*OP*_*SIZE*_*ARG*_MEM*MEM* X *Y
