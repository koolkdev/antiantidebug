OPTION RUN_ONCE
OPTION MATCH_ONE

DEFINE_GROUP ONE_ARG_OP PUSH POP INC DEC NEG NOT
DEFINE_GROUP TWO_ARG_OP MOV ADD SUB XOR AND OR SHL SHR ROL ROR RCL RCR CMP IMUL TEST XCHG
DEFINE_GROUP TWO_ARG_OP_SIZES MOVSX

DEFINE_GROUP ARGS VAR MEMVAR REG REGHIGH MEMREG IMM RELIMM

DEFINE_GROUP SIZES BYTE WORD DWORD
DEFINE_GROUP[64] SIZES QWORD

DEFINE_MACRO[32] SIZE DWORD
DEFINE_MACRO[64] SIZE QWORD

DEFINE_TEMPLATE
&OP=^ONE_ARG_OP|_|&SIZE=^SIZES|_REG SP
UPDATE_SP
=>
*OP*_*SIZE*_REG SP

DEFINE_TEMPLATE
&OP=^TWO_ARG_OP|_|&SIZE=^SIZES|_REG_|&ARG=^ARGS SP Y
UPDATE_SP
=>
*OP*_*SIZE*_REG_*ARG* SP Y

DEFINE_TEMPLATE
&OP=^TWO_ARG_OP_SIZES|_|&SIZE=^SIZES|_|&SIZE2=^SIZES|_REG_|&ARG=^ARGS SP Y
UPDATE_SP
=>
*OP*_*SIZE*_*SIZE2*_REG_*ARG* SP Y

# So it won't be deleted by the script later
DEFINE_TEMPLATE
MOV_BYTE_REG_REG AX AX
=>
_NOP

DEFINE_TEMPLATE
RETURN X:(X!=0)
=>
RET_IMM X

DEFINE_TEMPLATE
RETURN X:(X==0)
=>
RET

DEFINE_TEMPLATE
PUSH_@SIZE@_REG AX
END_VM
CALL_|&ARG=^ARGS X
=>
END_VM
CALL_*ARG* X
