OPTION RUN_ONCE

DEFINE_GROUP MOVS_STOS MOVS STOS
DEFINE_GROUP CMPS_SCAS CMPS SCAS

DEFINE_GROUP LETTER_SIZE B W D
DEFINE_GROUP[64] LETTER_SIZE Q

DEFINE_GROUP CC Z NZ

DEFINE_MACRO[32] SIZE DWORD
DEFINE_MACRO[64] SIZE QWORD

DEFINE_TEMPLATE
LABEL X
CMP_@SIZE@_REG_IMM CX 0x0
JZ Y
&OP=^MOVS_STOS|&SIZE=^LETTER_SIZE
DEC_@SIZE@_REG CX
JMP X
LABEL Y
=>
LABEL X
REP_*OP**SIZE*
LABEL Y

DEFINE_TEMPLATE
LABEL X
CMP_@SIZE@_REG_IMM CX 0x0
JZ Y
DEC_@SIZE@_REG CX
&OP=^CMPS_SCAS|&SIZE=^LETTER_SIZE
J|&J=^CC Y
JMP X
LABEL Y
=>
LABEL X
REP_*OP**SIZE*_*J*
LABEL Y

DEFINE_TEMPLATE
REP_|&OP=^CMPS_SCAS|&SIZE=^LETTER_SIZE|_NZ
=>
REPE_*OP**SIZE*

DEFINE_TEMPLATE
REP_|&OP=^CMPS_SCAS|&SIZE=^LETTER_SIZE|_Z
=>
REPNE_*OP**SIZE*

### Misc ###

DEFINE_TEMPLATE
MOV_@SIZE@_REG_REG SP BP
POP_@SIZE@_REG BP
=>
LEAVE

### PUSHA POPA ###

PUSH_DWORD_REG AX
PUSH_DWORD_REG CX
PUSH_DWORD_REG DX
PUSH_DWORD_REG BX
PUSH_DWORD_REG SP
ADD_DWORD_MEMREG_IMM SP 0x10
PUSH_DWORD_REG BP
PUSH_DWORD_REG SI
PUSH_DWORD_REG DI
=>
PUSHA

DEFINE_TEMPLATE[32]
POP_DWORD_REG DI
POP_DWORD_REG SI
POP_DWORD_REG BP
POP_DWORD_REG BX
POP_DWORD_REG BX
POP_DWORD_REG DX
POP_DWORD_REG CX
POP_DWORD_REG AX
=>
POPA
