# This file cleans some garbage
# In the vm code, there are groups of 
# PUSH_DX
# changing dx (all the groups of garbage)
# POP_DX

# Those are the instructions that are using to do the garbage, and don't use dx
DEFINE_GROUP NON_DX_RELATED STACK_PUSH_REGREF STACK_ADD PUSH_DWORD_IMM
DEFINE_GROUP[32] NON_DX_RELATED PUSH_WORD_IMM
DEFINE_GROUP[64] NON_DX_RELATED PUSH_QWORD_IMM

DEFINE_GROUP MATH_OPS_FLAGS ADD SUB XOR AND OR BT BTC BTR BTS IMULTWO ADC SBB CMP TEST INC DEC RCR RCL ROL ROR SAR SHL SHR MUL IMUL DIV IDIV NEG
DEFINE_GROUP SIMPLE_MATH ADD SUB XOR AND OR
DEFINE_GROUP SIMPLER_MATH ADD SUB XOR
DEFINE_GROUP NON_DX_RELATED ^SIMPLE_MATH|F_DWORD STACK_POP_DWORD_REG
DEFINE_GROUP CHANGE_DX STACK_POP_DX STACK_MOV_DX_SP
DEFINE_GROUP DX_OP ADD_DX SUB_DX XOR_DX

DEFINE_GROUP PUSHES STACK_PUSH_DX PUSH_JUNK

DEFINE_GROUP SIZES BYTE WORD DWORD
DEFINE_GROUP[64] SIZES QWORD

DEFINE_MACRO[32] SIZE_WORD DWORD
DEFINE_MACRO[32] SIZE_NUM 0x4
DEFINE_MACRO[64] SIZE_WORD QWORD
DEFINE_MACRO[64] SIZE_NUM 0x8


DEFINE_TEMPLATE_MACRO UNRELATED_POP_DX
*^NON_DX_RELATED
&I=^CHANGE_DX
